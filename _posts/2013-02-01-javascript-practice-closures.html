The inner function has access to the variables of the outer function. A closure is formed when a reference to the inner function exists outside the outer function and we can execute the inner function after the outer function returns, at this point the inner function still has access to the variables of the outer function. 
var outer = function () {

    var i = 0;

    return function () {
        alert(i++);
    };
};

var closure = outer();

closure(); // 0
closure(); // 1
closure(); // 2
Same example as above except the inner function is not returned, it has a global reference. 
var closure;

var outer = function () {

    var i = 0;

    closure =function () {
        alert(i++);
    };
};

outer();

closure(); // 0
closure(); // 1
closure(); // 2
